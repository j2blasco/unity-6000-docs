unityVersion; 
### Description
The version of the Unity runtime used to play the content.
```

public class Example : MonoBehaviour[](https://docs.unity3d.com/6000.0/Documentation/ScriptReference/MonoBehaviour.html)
{
    void Start()
    {
        print("Version of the runtime: " + Application.unityVersion[](https://docs.unity3d.com/6000.0/Documentation/ScriptReference/Application-unityVersion.html));
    }
}
```
